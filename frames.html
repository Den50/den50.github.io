Рыбалка

(fassert место_рыбалки
  (classification (generic))
  (место default(море) default(озеро) default(океан) default(река))
)
(fget место_рыбалки)

(fassert средства_ловли
  (classification (generic))
  (средства default(спиннинг) default(лодка) default(червяк) default(наживка))
)
(fget средства_ловли)

(fassert рыба
  (classification (generic))
  (вид default(хищная) default(полухищная) default(травоядная))
)
(fget рыба)

(fassert рыбалка_на_пресных_водоемах
  (classification (generic))
  (ako (value место_рыбалки))
  (ako (value средства_ловли))
  (ako (value рыба))
  (рыбак default(профессионал) default(любитель) default(спортсмен))
)
(fget рыбалка_на_пресных_водоемах)

(fassert любительская_рыбалка_на_реке_обь
  (classification (generic))
  (ako (value рыбалка_на_пресных_водоемах))
  (место (value(река_обь)))
  (рыбак (value(любитель)))
)
(fget любительская_рыбалка_на_реке_обь)
